File Upload Integration

// Component export class MyComponent { selectedFile: File | null = null; onFileSelected(event: Event) { const input = event.target as HTMLInputElement; if (input.files && input.files[0]) { this.selectedFile = input.files[0]; } } onDocumentError(error: string) { console.error('Error loading document:', error); } } // Template <input type="file" (change)="onFileSelected($event)" accept=".pdf,.jpg,.jpeg,.png,.gif,.bmp,.webp,.doc,.docx,.xls,.xlsx"> <app-document-viewer *ngIf="selectedFile" [src]="selectedFile" [fileName]="selectedFile.name" height="500px" (loadError)="onDocumentError($event)"> </app-document-viewer>
